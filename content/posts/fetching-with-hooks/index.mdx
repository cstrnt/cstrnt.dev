```jsx
function BlogPosts() {
  const [loading, setLoading] = useState(false);
  const [posts, setPosts] = useState(null);

  useEffect(() => {
      setLoading(true)
      fetch('SERVER-URL')
      .then(response => response.json())
      .then(data => {
        setPosts(data);
        setLoading(false)
      })
  }, []);

  return (
      ...
  )
}
```
